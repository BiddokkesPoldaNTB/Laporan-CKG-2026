function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); // Buat header jika baris pertama kosong if (sheet.getLastRow() === 0) { sheet.appendRow([ 'No','Timestamp','Nama Lengkap','NIK','Satker','NIP/NRP', 'Tanggal Lahir','Jenis Kelamin','No. WhatsApp', 'Status Nikah (Pendataan)','Pekerjaan','Alamat Domisili', 'Status Nikah (Demografi)','Status Disabilitas', 'Riwayat Batuk TB','Jumlah Foto','Link Foto' ]); } // Simpan foto ke Google Drive (jika ada) var linkFoto = '-'; if (data.fotoBase64) { try { var base64 = data.fotoBase64.split(',')[1]; // hapus prefix "data:image/...;base64," var mimeType = data.fotoBase64.split(';')[0].split(':')[1]; var blob = Utilities.newBlob(Utilities.base64Decode(base64), mimeType, data.namaFoto || 'foto.jpg'); // Simpan ke folder "Foto KTP CKG 2026" di Drive (akan dibuat otomatis jika belum ada) var folderName = 'Foto KTP CKG 2026'; var folders = DriveApp.getFoldersByName(folderName); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(folderName); var file = folder.createFile(blob); file.setName(data.nik + '_' + (data.namaFoto || 'foto.jpg')); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); linkFoto = file.getUrl(); } catch(fotoErr) { linkFoto = 'Gagal: ' + fotoErr.toString(); } } var no = sheet.getLastRow(); // nomor urut otomatis sheet.appendRow([ no, new Date().toLocaleString('id-ID'), data.namaLengkap, data.nik, data.satker, data.nipNrp, data.tanggalLahir, data.jenisKelamin, data.noWa, data.statusNikah, data.pekerjaan, data.alamat, data.statusNikahDemo, data.disabilitas, data.batukTB, data.jumlahFoto || 1, linkFoto ]); return ContentService .createTextOutput(JSON.stringify({ status: 'success', foto: linkFoto })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: 'Apps Script aktif' })) .setMimeType(ContentService.MimeType.JSON); }
